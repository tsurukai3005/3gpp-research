\begin{frame}{パラメータの依存構造}
\begin{itemize}
  \item Operating Band $\to$ サポートされる SCS を制約 (5.3.5)
  \item \textbf{SCS が分岐点}
    \begin{itemize}
      \item SCS $\times$ Channel BW $\to$ $N_{\mathrm{RB}}$ 決定 (5.3.2) $\to$ ガードバンド導出 (5.3.3)
      \item SCS $\to$ BWP のニューメロロジー選択肢を制約（BWP は単一 SCS）
    \end{itemize}
  \item Operating Band $\to$ Channel Raster $\Delta F_{\mathrm{Raster}}$ (5.4.2.3) $\to$ NR-ARFCN $\to$ Point A $\to$ CRB $\to$ BWP
  \item Operating Band $\to$ Synchronization Raster (5.4.3.3) $\to$ GSCN 範囲 + SSB SCS パターン
  \item Operating Band $\to$ TX-RX Separation (5.5, FDD のみ)
\end{itemize}
\end{frame}
